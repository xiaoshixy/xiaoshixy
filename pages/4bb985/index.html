#第一步：取得每个部门最高薪水（按照部门编号分组，找出每一组的最大值）
select deptno,max(sal) as maxsal from emp group by deptno;
#第二步：上述结果当成临时表 t,条件：t.deptno = e.deptno
select
    e.ename
from
    emp e
join
    (select deptno,max(sal) as maxsal from emp group by deptno) t
on
    e.deptno = t.deptno and e.sal = t.maxsal;

点击查看 
#第一步：找出每个部门的平均薪水
select deptno,avg(sal) as avgsal from emp group by deptno;
#第二步：上述结果当成临时表 t,条件：t.deptno = e.deptno 并且 e.sal > t.avgsal
select
    e.ename,
    e.sal
from
    emp e
join
    (select deptno,avg(sal) as avgsal from emp group by deptno) t

    e.deptno = t.deptno and e.sal > t.avgsal;

计算年差

  • 在mysql当中怎么计算两个日期的“年差”，差了多少年？

    • TimeStampDiff(间隔类型, 前一个日期, 后一个日期)
    • timestampdiff(YEAR, hiredate, now())

  • 间隔类型：

    SECOND 秒

    MINUTE 分钟

    HOUR 小时

    DAY 天

    WEEK 星期

    MONTH 月

    QUARTER 季度

    YEAR 年

点击查看 
select
	d.deptno, count(e.ename) ecount,ifnull(avg(e.sal),0) as avgsal, ifnull(avg(timestampdiff(YEAR, hiredate, now())), 0) as avgservicetime
from
	emp e
right join
	dept d
on
	e.deptno = d.deptno
group by
	d.deptno;
