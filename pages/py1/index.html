# 字符串截取

语法：str[起始位置：结束位置：步进值]

str1 = "iloveyou"
print(str1)
print(str1[0:-1])# 输出第一个到倒数第二个的所有字符
print(str1[0]) # 输出字符串第一个字符
print(str1[2:5]) # 输出从第三个开始到第五个的字符
print(str1[2:]) # 输出从第三个开始后的所有字符
print(str1 * 2) # 输出字符串两次
print(str1 + '你好') # 连接字符串
print(str1[:5]) # 输出第五个字母前的所有字符
print(str1[0:7:2])# [起始：终止：步长]
print('hello\nchengdu') # 使用反斜杠(\)+n转义特殊字符
print(r'hello\npython') # 在字符串前面添加一个 r，表示原始字符串，不会发生转义
